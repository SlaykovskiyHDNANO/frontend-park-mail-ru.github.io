Asynchronous Module Definition

            define('A', function () {
                var name = 'A'; // приватная переменная

                    }
                };
            });

Asynchronous Module Definition

            define('B', ['A'], function (A) {
                var name = 'B'; // приватная переменная
                return {
                    say: function () { // публичный метод
                        A.say(); // вызов публичного метода модуля A
                        alert(name);
                    }
                …
