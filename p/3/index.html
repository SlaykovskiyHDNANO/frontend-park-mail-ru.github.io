                         l = document.getElementsByTagName('div').length;
                     i < l;

                  document.getElementsByTagName('div')[i].innerHTML = i;
                }

Работа с DOM

                var nodes = document.getElementsByTagName('div');
                for (var i = 0,
                         l = nodes.length;
                     i < l;
                     i++) {
                  nodes[i].innerHTML = i;
                }
